% --- 1.4 Why crosswords ---
\begin{frame}{Why Crosswords?}
  \begin{columns}[c]
    \column{0.55\textwidth}
      \small
      A crossword is \textbf{structured gamification}:\\
      it forces active work inside a game.
      \vspace{0.6em}
      \begin{itemize}
        \setlength\itemsep{0.6em}
        \item \textbf{Active recall} $\rightarrow$ retention
        \item \textbf{Meaning-driven retrieval} $\rightarrow$ deeper learning
        \item \textbf{Immediate feedback} $\rightarrow$ self-assessment
        \item \textbf{The puzzle itself} $\rightarrow$ intrinsic motivation
      \end{itemize}
      \vspace{0.6em}
      {\scriptsize\color{graytext} Saxena et al.\ (2009); Patrick et al.\ (2018)}
    \column{0.45\textwidth}
      \centering
      \includegraphics[width=0.85\textwidth]{figures/crossword.pdf}
  \end{columns}
\end{frame}
