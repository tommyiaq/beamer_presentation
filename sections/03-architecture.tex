% --- 3.3 Grid generation ---
\begin{frame}{Building the Grid}
  \begin{columns}[c]
    \column{0.52\textwidth}
      \setlength{\fboxsep}{0.5em}
      \fcolorbox{unisired!40}{unisired!3}{\parbox{0.9\columnwidth}{%
        \scriptsize\ttfamily
        \textbf{\small\rmfamily Input:} clue--answer pairs\\[0.3em]
        place longest word at centre\\
        \textbf{while} stopping criteria not met \textbf{do}\\
        \hspace{1em}\textbf{if} all words placed \textbf{then}\\
        \hspace{2em}compute Score\\
        \hspace{2em}\textbf{if} Score $>$ best \textbf{then} save grid\\
        \hspace{1em}\textbf{else}\\
        \hspace{2em}find intersections for next word\\
        \hspace{2em}\textbf{if} placement found \textbf{then}\\
        \hspace{3em}place word in grid\\
        \hspace{2em}\textbf{else} backtrack, try alternative\\
        \textbf{end while}\\[0.3em]
        \textbf{\small\rmfamily Output:} best grid + clue-position map%
      }}
    \column{0.48\textwidth}
      \centering
      \vspace{0.5em}
      \textbf{Scoring function:}\\[0.6em]
      $\text{Score} = (FW + 0.5 \cdot LL) \times FR \times LR$\\[0.8em]
      {\small
      \begin{tabular}{@{}ll@{}}
        $FW$ & filled words\\
        $LL$ & linked (shared) letters\\
        $FR$ & filled ratio (compactness)\\
        $LR$ & linked letter ratio\\
      \end{tabular}}
  \end{columns}
\end{frame}
